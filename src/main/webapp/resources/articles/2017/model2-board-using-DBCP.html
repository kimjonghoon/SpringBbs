    articleno number,
    title varchar2(200) NOT NULL,
    content varchar2(4000),
    regdate date,
    parent number,
    constraint PK_HIERARCHY_ARTICLE PRIMARY KEY(articleno)
);

create sequence seq_hierarchy_article
increment by 1

테스트
