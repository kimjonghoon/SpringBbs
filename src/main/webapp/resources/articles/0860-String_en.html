String

var name = new String("John Doe");//or var name = "John Doe";
alert("A. Name: " + name);
alert("B. Name's Length: " + name.length);//자바는 length()
alert("C. " + name.charAt(1));//문자의 인덱스 1에 위치한 문자 반환,자바와 같음
alert("D. " + name.charCodeAt(1));//인덱스의 문자코드 반환, 자바의 codePointAt()
alert("E. " + name.concat(",Jane Doe,Anne Bonny"));//자바와 같음
alert("F. " + name.indexOf("Doe"));//길이라는 문자가 있는 인덱스 번지 반환,자바와 같음 
alert("G. " + name.lastIndexOf("Doe",0));//인덱스 0부터 왼쪽으로 검색,자바와 같음
alert("H. " + name.match(/Doe/));//정규표현식에 해당하는 문자열을 반환,자바엔 없음
alert("I. " + name.replace("Doe","Cusack"));//자바의 replaceAll()과 가장 비슷
alert("J. " + name.search(/Doe/g));//정규표현식에 해당하는 문자열의 찾아 인덱스 반환
alert("K. " + name.slice(1,2));//자바의 substring(int,int)
alert("L. " + name.slice(1));//자바의 substring(int)
alert("M. " + name.split(" "));//자바스크립트의 배열은,로 구분,자바의 split() 같음
alert("N. " + name.substring(1,2));//자바와 같음
alert("O. " + name.toUpperCase());//자바와 같음
alert("P. " + name.toLowerCase());//자바와 같음
